Example of usage

            
<aptos-logo></aptos-logo>

<script type="module">
    import {AptosLogo} from "./node_modules/@olton/aptos-logo/src/index.js";
</script>
<script>
    let current = "light"
    function toggle(){
        const logo = document.querySelector("aptos-logo")
        if (current === 'light') current = 'dark'; else current = 'light'
        logo.setAttribute('data-theme', current)
    }
</script>
